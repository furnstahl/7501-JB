from scipy.special import airy
from scipy.optimize import root

def Airy(x, index):
    """
    Airy function. Returns Ai, Ai', Bi, Bi'
    """
    ans = airy(x)
    return ans[index]

z_sol = root(Airy, -1, args=1)
print(f'{z_sol.x[0]:.10f}')
